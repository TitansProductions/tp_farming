4. Add on `server/tp-server_escrow_allow.lua` the new function for supporting hammer durability - even if you disable it, this must exist. Add the following / given function **AllowOnHammerBuildingUse** below **OnSeedPlacement**.

```lua
--- @param source : returns the player source id who used the hammer to build a scarecrow.
--- @param itemName : returns the used hammer item name.
AllowOnHammerBuildingUse = function(source, itemName, cb) -- version 2.3.3
    local _source = source 

    local HammerData = Config.Actions['SCARECROW_BUILDING'].Durability -- this returns from config the required durability options

    -- We check if this durability option is enabled.
    if HammerData.Enabled then 
        
        if API.getFramework() == 'tpzcore' then 

            /* tpz_inventory/item_registrations.lua

            ['ironhammer'] = {
                label = 'Iron Hammer',
                weight = 1.0,
                remove = 0,
                description = 'Tools',
                action = 'NONE',
                stackable = 0, -- <- must be 0 for unique
                droppable = 1,
                closeInventory = 1
            },*/

            local TPZInv   = exports.tpz_inventory:getInventoryAPI()
            local contents = TPZInv.getInventoryContents(_source)

            for _, item in pairs(contents) do 

                if item.item == itemName then 

                    if item.metadata.durability <= 0 then 
                        return cb(false)
                    end

                    -- @param item : requires the item name.
                    -- returns integer (0-1).
                    local isItemStackable = TPZInv.isItemStackable(itemName)

                    if isItemStackable == 0 then 
                        local randomValueRemove = math.random(HammerData.RemoveDurability.min, HammerData.RemoveDurability.max)
                        TPZInv.removeItemDurability(_source, itemName, randomValueRemove, item.itemId, HammerData.Delete)
                    end

                    break
                end

            end

            return cb(true)

        elseif API.getFramework() == 'vorp' then 

            local hammer <const> = exports.vorp_inventory:getItem(_source, itemName) -- item must be unique on vorp to have metadata.

            if not hammer then
                return cb(false)
            end

            local meta = hammer.metadata or {}

            -- If item has no durability metadata, initialize it at 100%
            if not meta.durability then 

                local randomValueRemove = math.random(HammerData.RemoveDurability.min, HammerData.RemoveDurability.max)
                local default_durability = 100 - randomValueRemove

                local metadata = { description = Config.Actions['SCARECROW_BUILDING'].DurabilityText .. " " .. default_durability .. '%', durability = default_durability }
                exports.vorp_inventory:setItemMetadata(_source, hammer.id, metadata, 1)

            else
                -- If durability exists, verify it is not already broken
                if meta.durability <= 0 then 
                    return cb(false)
                end

                local randomValueRemove = math.random(HammerData.RemoveDurability.min, HammerData.RemoveDurability.max)
                local durability = meta.durability - randomValueRemove
                local description = Config.Actions['SCARECROW_BUILDING'].DurabilityText .. " " .. durability .. '%'
                local metadata = { description = description, durability = durability }
        
                if durability <= 0 and HammerData.Delete then
                    exports.vorp_inventory:subItem(_source, itemName, 1, meta)
                else
                    exports.vorp_inventory:setItemMetadata(_source, hammer.id, metadata, 1)
                end

            end

            return cb(true)

        elseif API.getFramework() == 'rsg' then -- item must be unique on rsg to have metadata.
            local RSGCore = exports['rsg-core']:GetCoreObject() -- Core Getter
            local Player  = RSGCore.Functions.GetPlayer(_source)

            local svslot  = nil

            local randomValueRemove = math.random(HammerData.RemoveDurability.min, HammerData.RemoveDurability.max)

            for slot, itemData in pairs(Player.PlayerData.items) do

                if itemData.name == itemName then

                    local svslot = itemData.slot

                    if Player.PlayerData.items[svslot].info.quality == nil then
                        Player.PlayerData.items[svslot].info.quality = 100
                        Player.Functions.SetInventory(Player.PlayerData.items)
                    end
                    
                    local newquality = Player.PlayerData.items[svslot].info.quality - randomValueRemove
                    
                    Player.PlayerData.items[svslot].info.quality = newquality
                    Player.Functions.SetInventory(Player.PlayerData.items)

                    if newquality <= 0 then

                        Player.PlayerData.items[svslot].info.quality = 0
                        Player.Functions.SetInventory(Player.PlayerData.items)

                        if HammerData.Delete then
                            Player.Functions.RemoveItem(itemData.name, 1, svslot)
                        end

                    end
                    break
                    
                end
 
            end

            return cb(true)

        end

    else 
        return cb(true) -- no durability system
    end
end
```


## How to update?

- You can easily update the modified files by redownloading from the **Cfx.re Portal** and copy-paste on the mentioned directory file paths.
- If new options have been implemented to the configuration file or edited, **DO NOT** copy-paste the new configuration file, otherwise you will get the latest configuration but you will lose all the changes you have made. The proper way updating a configuration file is by finding the new configuration options or the modified ones and do it **manually**.

*Cfx.re Portal **DOES NOT** force-update your configuration file or any files, if it did, you would always lose all the configuration changes you made or would totally break the script, all files are updated or edited manually.*
