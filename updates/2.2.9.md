2. Add on the `server/tp-server_escrow_allow` the new function that is used for the durability:

```lua
--- @param source : returns the player source id who used the hoe / rake harvest.
--- @param itemName : returns the used hoe item name.
AllowOnRakeHarvest = function(source, itemName, cb) -- version 2.2.9
    local _source = source 

    -- Since we don't know what hoe item player has or used because for harvesting the system does not require
    -- from the item to be actually used but just being on inventory, we have to loop through inventory 
    -- contents and modify that item metadata durability.
    
    local RakeData = Config.Actions['RAKE_HARVEST'].Durability -- this returns from config the required durability options

    -- We check if this durability option is enabled.
    if RakeData.Enabled then 

        if API.getFramework() == 'tpzcore' then 

            /* tpz_inventory/item_registrations.lua

            ['hoe'] = {
                label = 'Garden Hoe',
                weight = 3.0,
                remove = 0,
                description = 'Tools',
                action = 'NONE',
                stackable = 0, -- <- must be 0 for unique
                droppable = 1,
                closeInventory = 1
            },*/

            local TPZInv   = exports.tpz_inventory:getInventoryAPI()
            local contents = TPZInv.getInventoryContents(_source)

            for _, item in pairs(contents) do 

                if item.item == itemName then 

                    if item.metadata.durability <= 0 then 
                        return cb(false)
                    end

                    -- @param item : requires the item name.
                    -- returns integer (0-1).
                    local isItemStackable = TPZInv.isItemStackable(itemName)

                    if isItemStackable == 0 then 
                        local randomValueRemove = math.random(RakeData.RemoveDurability.min, RakeData.RemoveDurability.max)
                        TPZInv.removeItemDurability(_source, itemName, randomValueRemove, item.itemId, RakeData.Delete)
                    end

                    break
                end

            end

            return cb(true)

        elseif API.getFramework() == 'vorp' then 

            local hoe <const> = exports.vorp_inventory:getItem(_source, itemName) -- item must be unique on vorp to have metadata.

            if not hoe then
                return cb(false)
            end

            local meta = hoe.metadata

            if meta.durability <= 0 then 
                return cb(false)
            end

            local randomValueRemove = math.random(RakeData.RemoveDurability.min, RakeData.RemoveDurability.max)
               
            if not next(meta) then

                local default_durability = 100

                default_durability = default_durability - randomValueRemove

                local metadata = { description = Config.Actions['RAKE_HARVEST'].DurabilityText .. " " .. default_durability .. '%', durability = default_durability }
                exports.vorp_inventory:setItemMetadata(_source, hoe.id, metadata, 1)

            else

                local durability = meta.durability - randomValueRemove
                local description = Config.Actions['RAKE_HARVEST'].DurabilityText .. " " .. durability .. '%'
                local metadata = { description = description, durability = durability }
        
                if durability <= 0 and RakeData.Delete then
                    exports.vorp_inventory:subItem(_source, itemName, 1, meta)
                else
                    exports.vorp_inventory:setItemMetadata(_source, hoe.id, metadata, 1)
                end

            end

            return cb(true)


        elseif API.getFramework() == 'rsg' then -- item must be unique on rsg to have metadata.
            local RSG      = exports['rsg-core']:GetCoreObject() -- Core Getter
            local xPlayer  = RSG.Functions.GetPlayer(_source)
            
            local contents = xPlayer.PlayerData.items

            for _, item in pairs(contents) do 

                if item.item == itemName or item.name == itemName then 

                    if item.unique == 1 then 
                        local randomValueRemove = math.random(RakeData.RemoveDurability.min, RakeData.RemoveDurability.max)
       
                        if item.info <= 0 then 
                            return cb(false)
                        end

                        local durability = (item.info.quality - randomValueRemove)

                        if durability <= 0 then 
                            durability = 0
                        end

                        local info = {
                            quality = durability,
                        }
                        
                        xPlayer.Functions.RemoveItem(itemName, 1, item.slot, 'removed-by-system')
                        
                        if (durability) > 0 or not RakeData.Delete then
                            xPlayer.Functions.AddItem(itemName, 1, item.slot, info, 'given-by-system')
                        end

                    end

                    break
                end

            end

            return cb(true)

        end


    else 
        return cb(true) -- no durability system
    end

end
```

3. Add on the `config.lua` where `"RAKE_HARVEST"` action is on `Config.Actions` the new data for the durability:

--- @param Enabled : Set to true if you want the hoe to have a durability (The hoe item must be a unique or not stackable depending on your framework).
--- @param RemoveDurability : The required durability value to be removed on every hoe use.
--- @param Delete : Set to true if you want the hoe to be completely removed from the inventory when durability hits 0.
Durability = {Enabled = true, RemoveDurability = { min = 5, max = 5}, Delete = false}, -- 2.2.9
DurabilityText = "Durability:", -- for vorp or rsg

## How to update?

- You can easily update the modified files by redownloading from the **Cfx.re Portal** and copy-paste on the mentioned directory file paths.
- If new options have been implemented to the configuration file or edited, **DO NOT** copy-paste the new configuration file, otherwise you will get the latest configuration but you will lose all the changes you have made. The proper way updating a configuration file is by finding the new configuration options or the modified ones and do it **manually**.

*Cfx.re Portal **DOES NOT** force-update your configuration file or any files, if it did, you would always lose all the configuration changes you made or would totally break the script, all files are updated or edited manually.*
